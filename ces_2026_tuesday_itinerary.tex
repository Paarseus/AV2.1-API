\section*{Pro Tips}
%----------------------------------------------------------

\begin{enumerate}[leftmargin=*, nosep]
    \item \textbf{Seat Strategy:} Arrive at W219 by 8:45 AM. After 9:40 Bosch session, walk 30 sec to W218 for prime seats before 10 AM crowd.
    \item \textbf{Atlas Timing:} 12 PM demo (lunch hour = fewer people), 5 PM demo (end of day = shorter lines).
    \item \textbf{Sessions are first-come-first-served} --- don't leave W218 between 10:40--11:00 AM.
    \item \textbf{Food:} West Hall concessions on Level 1, or Fontainebleau (5 min walk) for better options.
    \item \textbf{Skip North Hall Tuesday} --- K-Humanoid Alliance is 25 min walk. Hit it Wednesday.
    \item \textbf{Bring portable charger} --- 9+ hours on your feet.
\end{enumerate}
